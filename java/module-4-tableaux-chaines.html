    public static void analyserCaracteres(String texte) {
        int[] frequences = new int[26]; // a-z
        int voyelles = 0;
        int consonnes = 0;
        int chiffres = 0;
        int espacesEtPonctuation = 0;
        
        for (char c : texte.toLowerCase().toCharArray()) {
            if (c >= 'a' && c <= 'z') {
                frequences[c - 'a']++;
                if ("aeiou".indexOf(c) != -1) {
                    voyelles++;
                } else {
                    consonnes++;
                }
            } else if (c >= '0' && c <= '9') {
                chiffres++;
            } else {
                espacesEtPonctuation++;
            }
        }
        
        System.out.println("Voyelles: " + voyelles);
        System.out.println("Consonnes: " + consonnes);
        System.out.println("Chiffres: " + chiffres);
        System.out.println("Espaces et ponctuation: " + espacesEtPonctuation);
        
        // Affichage des lettres les plus frÃ©quentes
        System.out.println("\nLettres les plus frÃ©quentes:");
        for (int i = 0; i < 5; i++) { // Top 5
            int maxIndex = 0;
            for (int j = 1; j < 26; j++) {
                if (frequences[j] > frequences[maxIndex]) {
                    maxIndex = j;
                }
            }
            
            if (frequences[maxIndex] > 0) {
                char lettre = (char) ('a' + maxIndex);
                System.out.println("  " + lettre + ": " + frequences[maxIndex] + " fois");
                frequences[maxIndex] = 0; // Retirer pour le prochain
            }
        }
    }
    
    public static void analyserFrequenceMots(String[] mots) {

    public static double calculerIndiceLisibilite(int nbMots, int nbPhrases, double longueurMoyenne) {
        // Formule simplifiÃ©e inspirÃ©e de Flesch-Kincaid
        double motsParPhrase = (double) nbMots / nbPhrases;
        return 100 - (1.015 * motsParPhrase) - (84.6 * longueurMoyenne / nbMots * 100);
    }
    
    public static String detecterLangue(String texte) {
        String texteLower = texte.toLowerCase();
        
        // Mots indicateurs franÃ§ais
        String[] motsFrancais = {"le", "la", "les", "de", "du", "des", "et", "est", "une", "que", "avec"};
        int scoreFrancais = 0;
        
        // Mots indicateurs anglais
        String[] motsAnglais = {"the", "and", "is", "are", "with", "that", "this", "for", "you", "have"};
        int scoreAnglais = 0;
        
        for (String mot : motsFrancais) {
            if (texteLower.contains(" " + mot + " ") || texteLower.startsWith(mot + " ")) {
                scoreFrancais++;
            }
        }
        
        for (String mot : motsAnglais) {
            if (texteLower.contains(" " + mot + " ") || texteLower.startsWith(mot + " ")) {
                scoreAnglais++;
            }
        }
        
        if (scoreFrancais > scoreAnglais) {
            return "FranÃ§ais (score: " + scoreFrancais + ")";
        } else if (scoreAnglais > scoreFrancais) {
            return "Anglais (score: " + scoreAnglais + ")";
        } else {
            return "IndÃ©terminÃ©";
        }
    }
