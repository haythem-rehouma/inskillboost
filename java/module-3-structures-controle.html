    public static int rechercheLineaire(int[] tableau, int element) {

            }

    public static int[] trouverMinMax(int[] tableau) {
        if (tableau.length == 0) {
            return new int[]{0, 0};
        }
        
        int min = tableau[0];
        int max = tableau[0];
        
        for (int i = 1; i < tableau.length; i++) {
            if (tableau[i] < min) {
                min = tableau[i];
            }
            if (tableau[i] > max) {
                max = tableau[i];
            }
        }
        
        return new int[]{min, max};
    }

        double somme = 0;

            mediane = tableauTrie[milieu];

            // Si aucun √©change, le tableau est tri√©

            }
        }

        double base = scanner.nextDouble();

        if (exposant >= 0) {
            for (int i = 0; i < exposant; i++) {

            for (int i = 0; i > exposant; i--) {
