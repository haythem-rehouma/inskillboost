        }
    }

            if (sortConnusus[i].equals(nomSort)) {
                sortConnu = true;
                break;
            }

        if (!sortConnu) {
            System.out.println(nom + " ne conna√Æt pas le sort: " + nomSort);
            return;

                coutMana = 25;
                degats = 30 + intelligence;
                break;
            case "√âclair":

                degats = 25 + intelligence;
                break;
            case "Bouclier Magique":
                coutMana = 15;
                degats = 0; // Sort d√©fensif
                break;
        }

            manaActuel -= coutMana;
            System.out.println("üîÆ " + nom + " lance " + nomSort + " sur " + cible.getNom());

                System.out.println("‚ú® Bouclier magique activ√© pour " + nom);
            }

        } else {
            System.out.println("Mana insuffisant pour " + nomSort + " (Co√ªt: " + coutMana + ", Actuel: " + manaActuel + ")");
        }
    }
